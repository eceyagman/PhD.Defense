\section{Introduction}
% ---- 1: The big picture ----
\begin{frame}{The big picture}
\footnotesize
\fadecolor{1}{\textbf{Skills beyond cognitive abilities matter}}
\vspace{0.1cm}
\begin{itemize}
    \item[\fadecolor{1}{\textbullet}] \fadecolor{1}{\textbf{Socioemotional skills:} the umbrella term for maintaining relationships, regulating emotions, managing goals (e.g., perseverance, teamwork, self-control, etc.)}
    \item[\fadecolor{1}{\textbullet}] \fadecolor{1}{Predict educational attainment, earnings, well-being, long-term success \citep{Heckman2012-nt}; malleable esp. during adolescence}
\end{itemize}
\vspace{0.1cm}
\fadecolor{2}{\textbf{But, how do we measure them?}}
\begin{itemize}
    \item[\fadecolor{2}{\textbullet}] \fadecolor{2}{Lack of common language: competencies, soft skills, non-cognitive skills, etc.}
    \item[\fadecolor{2}{\textbullet}] \fadecolor{2}{Lack of precision: 136  frameworks identified}
    \item[\fadecolor{2}{\textbullet}] \fadecolor{2}{Assessment tools suffer from biases, reliability, and validity concerns}
\end{itemize}
\vspace{0.1cm}
\fadecolor{3}{\textbf{The challenge is especially acute in schools:}}
\begin{itemize}
    \item[\fadecolor{3}{\textbullet}] \fadecolor{3}{Education systems are increasingly expected to foster socioemotional learning (SEL), not just academics}
    \item[\fadecolor{3}{\textbullet}] \fadecolor{3}{Daily classroom interactions, instructional practices, teacher feedback and expectations shape their development}
    \item[\fadecolor{3}{\textbullet}] \fadecolor{3}{Yet educators lack consistent tools for developing and tracking these skills}
\end{itemize}
\end{frame}
% ---- 2: This thesis (preview) ----
\begin{frame}{This thesis - a preview}
\small
\vspace{0.3cm}
\textcolor{HopBlue}{\textbf{What happens when structured SEL is embedded into routine classroom instruction?}} \\
\vspace{0.4cm}
\begin{itemize}
    \item[\fadecolor{1}{$\blacktriangleright$}] \fadecolor{1}{\textbf{A pedagogical program} that trains teachers to observe, assess, and give feedback on socioemotional skills}
    \item[\fadecolor{2}{$\blacktriangleright$}] \fadecolor{2}{\textbf{Multiple measurement approaches} (classroom observations, behavioral tasks, surveys, admin records)}
    \item[\fadecolor{3}{$\blacktriangleright$}] \fadecolor{3}{\textbf{A randomized experiment} in 40 secondary schools in Catalonia}
\end{itemize}
\vspace{0.5cm}

\fadecolor{4}{\textbf{Three questions:}}
\begin{enumerate}
    \item[\fadecolor{4}{1.}] \fadecolor{4}{Does the program change classrooms and student outcomes?}
    \item[\fadecolor{4}{2.}] \fadecolor{4}{Are the evaluations it generates gender-neutral?}
    \item[\fadecolor{4}{3.}] \fadecolor{4}{Can the program reduce in-group favoritism?}
\end{enumerate}
\end{frame}

% ---- 6: Multi-rater assessment ----
\begin{frame}{Multi-rater assessment}\label{slide:classroom}
\small
\textbf{After each activity:} Multi-rater scoring on a 5-point scale
\begin{itemize}
    \item Teachers rate students; Students rate themselves and group peers (360$^{\circ}$ evaluation)
    \item Ratings are confidential (not tied to formal grades); used to generate feedback reports
    \item External observers (8--10 per classroom) provide independent ratings for research only
\end{itemize}
\vspace{0.3cm}
\textcolor{HopBlue}{\textbf{Key design feature:}} A common behavioral rubric enables direct comparisons across rater types.
\end{frame}

% ---- 14: Econometric specification ----
\begin{frame}{Econometric specification}
\small
Student-level or teacher-level regressions estimating intent-to-treat (ITT) effects:
\begin{equation*}
y_{igst} = \alpha + \beta T_{igs} + \gamma y_{igs0}
  + \sum_{w=1}^{W} \delta_i^w + \sum_{c=1}^{4} \eta_s^c
  + \mu_s + \varepsilon_{igst}
\end{equation*}
\begin{itemize}
\footnotesize
    \item $y_{igs0}$: Outcome at baseline \textit{(when available)}
    \item $\mu_s$: School fixed effects
    \item $\delta$: school disadvantage level strata, $\eta$: week-of-the-year of measurement FE
    \item Standard errors clustered at the school-grade level \citep{Abadie2022-xz}
\end{itemize}
\end{frame}

% ---- 25: Why Adolescence? ----
\begin{frame}{Adolescence as a formative period}
\small
\begin{itemize}
    \item Ages 12--16: high neuroplasticity, self-perception and aspirations are particularly malleable \citep{Steinberg2014-pe}
    \item \textbf{Self-identity formation:} Who am I? What am I good at? Where do I fit in?
    \item \textbf{Peer-influence peaks:} Students look to classmates for cues; who is good at what?
    \item Gendered norms around ``girls do well'' vs.\ ``boys do well'' solidify during these years
\end{itemize}
\vspace{0.5cm}
\textit{Early differences in self-perceptions or peer recognition can shape confidence and aspirations for years to come}
\end{frame}

\begin{frame}{Chapter 3: Takeaway}
\small
\textbf{A pedagogy designed for skill-building can reduce in-group favoritism as a secondary effect.}
\vspace{0.3cm}
\textbf{Three key results:}
\begin{enumerate}
    \item \textbf{Gender:} treatment compresses in-group premium in task-based settings (especially Cooperation)
    \item \textbf{Nationality:} treatment effects concentrated in low-diversity classrooms, where cross-group contact is most constrained
    \item \textbf{FA vs.\ BESSI:} task-based ratings update first (anchored in direct behavior); survey-based beliefs are stickier
\end{enumerate}
\vspace{0.3cm}
\textbf{Two channels:}
\begin{itemize}
    \item \textcolor{HopBlue}{\textbf{Cooperative contact:}} rotating groups create structured, equal-status intergroup interaction (Allport's conditions)
    \item \textcolor{HopBlue}{\textbf{Disciplined evaluation:}} behavior-anchored rubrics replace priors with observable evidence
\end{itemize}
\vspace{0.2cm}
\textit{Debiasing emerges as a byproduct of SEL pedagogy}

\begin{frame}{Gender Interaction}
\label{slide:gender_int}
\footnotesize
\textbf{Peer-evaluation regression:}
\[
y = \alpha + \beta_1\, \text{StuFem} + \beta_2\, \text{PeerFem} + \beta_3\, (\text{StuFem} \times \text{PeerFem}) + \ldots
\]
\medskip
\textbf{Female peer's rating of a \textcolor{blue!70!black}{female} vs.\ \textcolor{blue!70!black}{male} student:}
\begin{align*}
\text{Female peer -- Female student:} &\quad \beta_1 + \beta_2 + \beta_3 \\
\text{Female peer -- Male student:}   &\quad \beta_2 \\
\text{Difference:}     &\quad (\beta_1 + \beta_2 + \beta_3)\ -\ (\beta_2) = \bm{\beta_1 + \beta_3}
\end{align*}
\medskip
\textbf{So:}
\begin{itemize}
    \item[] $\bm{\beta_1 + \beta_3}$ captures the additional effect of being a \textbf{female student} (vs.\ male student) \emph{when the peer evaluator is female}.
    \item[] This is the ``female peer $\rightarrow$ female student premium.''
\end{itemize}

\end{frame}
