\section{Introduction} 

% ---- Slide 2: The Big Picture ----
\begin{frame}{The Big Picture}

\textbf{Skills beyond cognitive abilities matter}

\vspace{0.2cm}
\begin{itemize}
    \item \textbf{Socioemotional skills (SES):} the umbrella term for maintaining relationships, regulating emotions, managing goals
    \item e.g., perseverance, motivation, teamwork, self-control
    \item Predict educational attainment, earnings, well-being \citep{Heckman2012-nt}; malleable
\end{itemize}

\vspace{0.3cm}
\textbf{But, how do we measure them?}
\begin{itemize}
    \item Lack of common language: 21st-century skills, soft skills, non-cognitive skills, etc.
    \item Lack of precision: 136 competency frameworks identified
    \item Assessment tools suffer from biases, reliability, and validity concerns
\end{itemize}

\vspace{0.3cm}
\textbf{The challenge is especially acute in schools:}
\begin{itemize}
    \item Education systems are increasingly expected to foster SES, not just academics
    \item Daily classroom interactions shape their development
    \item Yet educators lack consistent tools for tracking these skills
\end{itemize}
\end{frame}


% ---- Slide 3: This Thesis ----
\begin{frame}{This Thesis}

\textbf{What happens when SEL is embedded into routine classroom instruction?}

\vspace{0.5cm}
\emph{Pentabilities}: 35 observable behaviors $\times$ 5 domains $\times$ repeated 360$^{\circ}$ assessment (teacher, peer, self) via digital platform, to structure feedback and goal setting.

\vspace{0.5cm}
\textbf{Three questions:}
\begin{enumerate}
    \item Does the pedagogy change classroom practices and student outcomes?
    \item Do systematic gender patterns emerge in self- and peer-evaluations?
    \item Can the program reduce identity-based evaluative gaps?
\end{enumerate}
\end{frame}

\section{Chapter 2}

% ---- Slide 18: Why Adolescence? ----
\begin{frame}{Why Adolescence?}

\begin{itemize}
    \item Ages 12--16: self-beliefs and aspirations are particularly malleable \citep{Steinberg2014-pe}
    \item \textbf{Self-identity formation:} Who am I? What am I good at?
    \item \textbf{Peer influence peaks:} Classmates become key reference point
    \item Gendered norms solidify during these years
\end{itemize}

\vspace{0.5cm}
$\Rightarrow$ Early gaps in self-beliefs can compound over time
\end{frame}


% ---- Slide 19: Two Research Questions ----
\begin{frame}{Two Research Questions}
\small
\textbf{Q1: Self-evaluation}\\
Do females rate their own SES differently, conditional on ability?

\vspace{0.5cm}
\textbf{Q2: Peer evaluation}\\
Is there a systematic gender pattern in how peers evaluate each other?

\vspace{0.5cm}
Evaluations shape beliefs, motivation, and feedback $\rightarrow$ gendered patterns in adolescence can influence later choices
\end{frame}


% ---- Slide 20: Empirical Strategy — Self-Evaluation ----
\begin{frame}{Empirical Strategy: Self-Evaluation}
\begin{block}{Specification (CO, BESSI, FA)}
\vspace{-0.2cm}
\begin{equation*}
\text{Self}_{is} = \alpha
   + \boldsymbol{\gamma_1} \mathbf{Female}_i
   + \gamma_2 \, \text{Teacher}_{is}

\end{frame}

\textbf{Control:} FF premium +0.29 to +0.43 SD (all significant), larger than FA

\vspace{0.4cm}
\textbf{Treatment:} Negative in 4/5 domains ($-$0.02 to $-$0.16 SD) but \textbf{not significant}

\vspace{0.5cm}
Same direction as FA but less precise. Reputation-based beliefs are stickier
\end{frame}

% ---- Slide 38: Two Mechanisms ----
\begin{frame}{Two Mechanisms}

\textbf{1. Cooperative contact}
\begin{itemize}
    \item Rotating groups $\rightarrow$ expanded intergroup interaction
    \item Equal-status, goal-directed collaboration (Allport, 1954)
\end{itemize}

\vspace{0.5cm}
\textbf{2. Disciplined evaluation}
\begin{itemize}
    \item Behavior-anchored rubrics $\rightarrow$ concrete criteria
    \item Ratings as formative data, not snap judgments
\end{itemize}

\vspace{0.5cm}
Changes both WHO you interact with and HOW you evaluate them
\end{frame}


% ---- Slide 39: Chapter 3 Takeaway ----

\textbf{A skill-building pedagogy can reduce in-group favoritism}

\vspace{0.4cm}
\begin{enumerate}
    \item \textbf{Gender:} Treatment compresses FF premium (especially Cooperation)
    \vspace{0.2cm}
    \item \textbf{Nationality:} Effects emerge in low-diversity classrooms
    \vspace{0.2cm}
    \item \textbf{FA vs.\ BESSI:} Behavioral ratings update faster than reputational beliefs
\end{enumerate}
\end{frame}

\end{itemize}
